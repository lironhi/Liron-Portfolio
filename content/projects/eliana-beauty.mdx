```prisma
model User {
  id        String   @id @default(uuid())
  email     String   @unique
  password  String

  firstName String
  lastName  String
  phone     String?
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  deletedAt DateTime? // Soft deletes

  appointments Appointment[]
}

model Service {
  id          String   @id @default(uuid())
  nameEn      String
  nameHe      String
  descriptionEn String?
  descriptionHe String?
  duration    Int      // minutes
  price       Decimal
  image       String?
  categoryId  String
  category    Category @relation(fields: [categoryId])
  staff       Staff[]
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  deletedAt   DateTime?
}

model Staff {
  id          String         @id @default(uuid())
  userId      String         @unique
  user        User           @relation(fields: [userId])
  services    Service[]
  workingHours WorkingHours[]
  timeOffs    TimeOff[]
  appointments Appointment[]
}

model Appointment {
  id        String            @id @default(uuid())
  clientId  String
  client    User              @relation(fields: [clientId])
  staffId   String
  staff     Staff             @relation(fields: [staffId])
  serviceId String
  service   Service           @relation(fields: [serviceId])
  date      DateTime
  startTime String
  endTime   String
  status    AppointmentStatus @default(PENDING)
  price     Decimal           // Preserves price at booking time
  notes     String?
  createdAt DateTime          @default(now())
  updatedAt DateTime          @updatedAt

  @@index([date, staffId]) // Optimized for availability queries
  @@index([clientId])      // Optimized for client history
}

model WorkingHours {
  id         String  @id @default(uuid())
  staffId    String
  staff      Staff   @relation(fields: [staffId])
  dayOfWeek  Int     // 0 = Sunday, 6 = Saturday
  startTime  String  // "09:00"
  endTime    String  // "17:00"
  isActive   Boolean @default(true)
}

model TimeOff {
  id        String      @id @default(uuid())
  staffId   String
  staff     Staff       @relation(fields: [staffId])
  type      TimeOffType // VACATION | SICK | OTHER
  startDate DateTime
  endDate   DateTime
  reason    String?
  createdAt DateTime    @default(now())
}
```

**Key Database Features:**
- **Soft deletes**: Data integrity with `deletedAt` field
- **Timestamp tracking**: Automatic `createdAt`/`updatedAt`
- **Price preservation**: Appointment prices frozen at booking time
- **Index optimization**: Strategic indexes for query performance
- **Type safety**: Prisma ensures compile-time type checking

### Availability Algorithm

```typescript
class AvailabilityEngine {
  generateSlots(date: Date, serviceId: string, staffId: string) {
    // 1. Get staff working hours for day
    const workingHours = this.getWorkingHours(staffId, date);

    // 2. Get existing appointments
    const appointments = this.getAppointments(staffId, date);

    // 3. Get time-offs
    const timeOffs = this.getTimeOffs(staffId, date);

    // 4. Get service duration
    const duration = this.getServiceDuration(serviceId);

    // 5. Generate available slots
    const slots = this.calculateSlots(
      workingHours,
      appointments,
      timeOffs,
      duration
    );

    // 6. Color-code slots
    return slots.map(slot => ({
      time: slot,
      status: this.getSlotStatus(slot, duration),
      color: this.getColorCode(slot)
    }));
  }

  getSlotStatus(slot: TimeSlot, duration: number): SlotStatus {
    // Red: Insufficient time until end of day/next appointment
    // Grey: Already booked or in time-off period
    // Green: Available with sufficient time
  }
}
```
