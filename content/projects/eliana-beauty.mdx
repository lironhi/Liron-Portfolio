# Eliana Beauty - Salon Booking Platform

A modern web application designed for beauty salons to manage appointments, services, and customer relationships with an intuitive user interface and real-time availability checking.

## 🎯 Problem & Solution

**Problem**: Traditional beauty salons rely on phone calls and manual scheduling, leading to double bookings, missed appointments, and poor customer experience. Customers struggle to see available time slots and service options.

**Solution**: An online booking platform that displays real-time availability, allows customers to browse services, book appointments 24/7, and manage their booking history—all from a user-friendly web interface.

### 1. Interactive Booking Calendar
```javascript
class BookingCalendar {
  constructor(containerId, options = {}) {
    this.container = document.getElementById(containerId);
    this.selectedDate = new Date();
    this.availableSlots = [];
    this.bookedSlots = [];
    this.init();
  }

  init() {
    this.renderCalendar();
    this.attachEventListeners();
    this.fetchAvailability();
  }

  renderCalendar() {
    const daysInMonth = new Date(
      this.selectedDate.getFullYear(),
      this.selectedDate.getMonth() + 1,
      0
    ).getDate();

    let html = '<div class="calendar-grid">';

    // Render day headers
    const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
    dayNames.forEach(day => {
      html += `<div class="calendar-day-header">${day}</div>`;
    });

    // Render dates
    const firstDayOfMonth = new Date(
      this.selectedDate.getFullYear(),
      this.selectedDate.getMonth(),
      1
    ).getDay();

    // Empty cells before first day
    for (let i = 0; i < firstDayOfMonth; i++) {
      html += '<div class="calendar-day empty"></div>';
    }

    // Calendar days
    for (let day = 1; day <= daysInMonth; day++) {
      const date = new Date(
        this.selectedDate.getFullYear(),
        this.selectedDate.getMonth(),
        day
      );
      const isToday = this.isToday(date);
      const isPast = date < new Date().setHours(0, 0, 0, 0);
      const hasAvailability = this.checkAvailability(date);

      html += `
        <div class="calendar-day ${isToday ? 'today' : ''} ${isPast ? 'past' : ''} ${hasAvailability ? 'available' : 'unavailable'}"
             data-date="${date.toISOString()}">
          <span class="day-number">${day}</span>
          ${hasAvailability ? '<span class="availability-indicator"></span>' : ''}
        </div>
      `;
    }

    html += '</div>';
    this.container.innerHTML = html;
  }

  checkAvailability(date) {
    // Check if date has available time slots
    const dateStr = date.toISOString().split('T')[0];
    return this.availableSlots.some(slot => slot.date === dateStr && !slot.booked);
  }

  async fetchAvailability() {
    try {
      const response = await fetch('/api/availability', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          month: this.selectedDate.getMonth() + 1,
          year: this.selectedDate.getFullYear()
        })
      });

      const data = await response.json();
      this.availableSlots = data.slots;
      this.renderCalendar();
    } catch (error) {
      console.error('Failed to fetch availability:', error);
    }
  }
}
```

```javascript
class ServiceCatalog {
  constructor() {
    this.services = [];
    this.selectedService = null;
    this.init();
  }

  async loadServices() {
    try {
      const response = await fetch('/api/services');
      this.services = await response.json();
      this.renderServices();
    } catch (error) {
      console.error('Failed to load services:', error);
    }
  }

  renderServices() {
    const container = document.getElementById('services-container');

    const html = this.services.map(service => `
      <div class="service-card" data-service-id="${service.id}">
        <div class="service-image">
          <img src="${service.image}" alt="${service.name}" loading="lazy">
        </div>
        <div class="service-content">
          <h3 class="service-name">${service.name}</h3>
          <p class="service-description">${service.description}</p>
          <div class="service-details">
            <span class="service-duration">
              <i class="icon-clock"></i> ${service.duration} min
            </span>
            <span class="service-price">
              <i class="icon-tag"></i> ₪${service.price}
            </span>
          </div>
          <button class="btn-book" onclick="selectService(${service.id})">
            Book Now
          </button>
        </div>
      </div>
    `).join('');

    container.innerHTML = html;
  }

  selectService(serviceId) {
    this.selectedService = this.services.find(s => s.id === serviceId);

    // Update UI to show selected state
    document.querySelectorAll('.service-card').forEach(card => {
      card.classList.remove('selected');
    });
    document.querySelector(`[data-service-id="${serviceId}"]`).classList.add('selected');

    // Move to time slot selection
    this.showTimeSlots();
  }

  showTimeSlots() {
    const timeSlotsSection = document.getElementById('time-slots');
    timeSlotsSection.style.display = 'block';
    timeSlotsSection.scrollIntoView({ behavior: 'smooth' });
  }
}
```

### 3. Booking Form with Validation
```javascript
class BookingForm {
  constructor(formId) {
    this.form = document.getElementById(formId);
    this.errors = {};
    this.attachValidation();
  }

  attachValidation() {
    this.form.addEventListener('submit', (e) => {
      e.preventDefault();
      if (this.validate()) {
        this.submitBooking();
      }
    });

    // Real-time validation
    this.form.querySelectorAll('input, textarea').forEach(field => {
      field.addEventListener('blur', () => {
        this.validateField(field);
      });
    });
  }

  validate() {
    this.errors = {};

    const name = this.form.querySelector('#name').value.trim();
    const phone = this.form.querySelector('#phone').value.trim();
    const email = this.form.querySelector('#email').value.trim();

    // Name validation
    if (!name || name.length < 2) {
      this.errors.name = 'Please enter your full name';
    }

    // Phone validation (Israeli format)
    const phoneRegex = /^05\d{8}$/;
    if (!phoneRegex.test(phone.replace(/[-\s]/g, ''))) {
      this.errors.phone = 'Please enter a valid Israeli phone number (05X-XXXXXXX)';
    }

    // Email validation
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (email && !emailRegex.test(email)) {
      this.errors.email = 'Please enter a valid email address';
    }

    // Display errors
    this.displayErrors();

    return Object.keys(this.errors).length === 0;
  }

  validateField(field) {
    const errorElement = document.querySelector(`#${field.id}-error`);

    if (this.errors[field.id]) {
      field.classList.add('error');
      if (errorElement) {
        errorElement.textContent = this.errors[field.id];
        errorElement.style.display = 'block';
      }
    } else {
      field.classList.remove('error');
      if (errorElement) {
        errorElement.style.display = 'none';
      }
    }
  }

  displayErrors() {
    Object.keys(this.errors).forEach(fieldId => {
      const field = this.form.querySelector(`#${fieldId}`);
      if (field) {
        this.validateField(field);
      }
    });
  }

  async submitBooking() {
    const formData = new FormData(this.form);
    const bookingData = {
      service_id: formData.get('service_id'),
      date: formData.get('date'),
      time: formData.get('time'),
      customer: {
        name: formData.get('name'),
        phone: formData.get('phone'),
        email: formData.get('email'),
        notes: formData.get('notes')
      }
    };

    try {
      const response = await fetch('/api/bookings', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(bookingData)
      });

      if (response.ok) {
        const result = await response.json();
        this.showConfirmation(result);
      } else {
        const error = await response.json();
        this.showError(error.message);
      }
    } catch (error) {
      this.showError('Failed to create booking. Please try again.');
    }
  }

  showConfirmation(booking) {
    const modal = document.getElementById('confirmation-modal');
    modal.querySelector('.booking-id').textContent = booking.id;
    modal.querySelector('.booking-date').textContent = booking.date;
    modal.querySelector('.booking-time').textContent = booking.time;
    modal.querySelector('.service-name').textContent = booking.service.name;
    modal.style.display = 'block';
  }
}
```
