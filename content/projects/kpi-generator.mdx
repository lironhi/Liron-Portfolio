# KPI Generator - Sales Performance Dashboard

A Python-based tool designed to automate the generation of Key Performance Indicator (KPI) reports for sales teams, providing comprehensive insights into sales performance with visual analytics.

## üéØ Problem & Solution

**Problem**: Sales managers spend hours manually compiling sales data from various sources, calculating KPIs, and creating reports. This manual process is time-consuming, error-prone, and delays decision-making.

**Solution**: An automated KPI generator that consolidates sales data, calculates relevant metrics, and produces professional reports with visualizations‚Äîreducing report generation time from hours to minutes.

## üîë Key Features

### 1. Automated Data Processing
```python
import pandas as pd
from datetime import datetime, timedelta

class SalesDataProcessor:
    def __init__(self, data_source):
        self.data = pd.read_csv(data_source)
        self.processed_data = None

    def clean_data(self):
        """Clean and prepare sales data"""
        # Remove duplicates
        self.data.drop_duplicates(subset=['transaction_id'], inplace=True)

        # Convert dates
        self.data['date'] = pd.to_datetime(self.data['date'])

        # Handle missing values
        self.data['amount'].fillna(0, inplace=True)

        return self

    def calculate_metrics(self):
        """Calculate key performance metrics"""
        self.processed_data = {
            'total_revenue': self.data['amount'].sum(),
            'avg_transaction': self.data['amount'].mean(),
            'transaction_count': len(self.data),
            'top_products': self.get_top_products(),
            'sales_by_region': self.get_regional_breakdown(),
            'growth_rate': self.calculate_growth_rate()
        }
        return self.processed_data
```

### 4. Report Generation
```python
from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer, Image, Table

class ReportGenerator:
    def __init__(self, kpi_data, output_path):
        self.data = kpi_data
        self.output_path = output_path
        self.doc = SimpleDocTemplate(output_path, pagesize=letter)
        self.story = []

    def generate_executive_summary(self):
        """Create executive summary section"""
        summary = f"""
        <b>Executive Summary</b><br/>
        <br/>
        Total Revenue: ${self.data['total_revenue']:,.2f}<br/>
        Growth Rate: {self.data['growth_rate']:.1f}%<br/>
        Transactions: {self.data['transaction_count']:,}<br/>
        Average Order Value: ${self.data['avg_transaction']:.2f}<br/>
        """
        return Paragraph(summary)

    def add_chart(self, chart_figure, title):
        """Add chart to report"""
        # Save chart as image
        img_path = f"temp_chart_{title}.png"
        chart_figure.savefig(img_path, dpi=300, bbox_inches='tight')

        # Add to report
        self.story.append(Spacer(1, 12))
        self.story.append(Paragraph(f"<b>{title}</b>"))
        self.story.append(Image(img_path, width=500, height=300))

    def create_kpi_table(self):
        """Create KPI summary table"""
        data = [
            ['KPI', 'Current', 'Previous', 'Change'],
            ['Revenue', f"${self.data['revenue']:,.0f}",
             f"${self.data['prev_revenue']:,.0f}",
             f"{self.data['revenue_change']:+.1f}%"],
            ['Customers', f"{self.data['customers']:,}",
             f"{self.data['prev_customers']:,}",
             f"{self.data['customer_change']:+.1f}%"],
            ['Avg Order', f"${self.data['avg_order']:.2f}",
             f"${self.data['prev_avg_order']:.2f}",
             f"{self.data['order_change']:+.1f}%"],
        ]

        table = Table(data)
        table.setStyle([
            ('BACKGROUND', (0, 0), (-1, 0), '#3b82f6'),
            ('TEXTCOLOR', (0, 0), (-1, 0), 'white'),
            ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
            ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
            ('GRID', (0, 0), (-1, -1), 1, 'black')
        ])

        return table

    def build_report(self):
        """Compile and save the complete report"""
        self.story.append(self.generate_executive_summary())
        self.story.append(Spacer(1, 20))
        self.story.append(self.create_kpi_table())

        # Add charts
        charts = self.generate_all_charts()
        for title, chart in charts.items():
            self.add_chart(chart, title)

        # Build PDF
        self.doc.build(self.story)
```
