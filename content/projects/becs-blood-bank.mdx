# BECS - Blood Bank Management System

A comprehensive web-based blood bank management system designed to streamline blood donation operations, track inventory, manage donors, and coordinate blood distribution to hospitals and medical facilities.

## üéØ Problem & Solution

**Problem**: Blood banks face complex challenges in managing donor information, tracking blood inventory by type, monitoring expiration dates, and coordinating distribution to medical facilities. Manual systems lead to waste, shortages, and inefficient operations.

**Solution**: BECS (Blood Bank Electronic Control System) provides an integrated platform that automates inventory management, donor tracking, blood testing records, and distribution logistics‚Äîensuring efficient operations and reducing blood wastage.

### 1. Inventory Management
```javascript
class BloodInventory {
  constructor() {
    this.inventory = new Map();
    this.expirationAlerts = [];
  }

  addBloodUnit(unit) {
    const { bloodType, donationDate, expirationDate, unitId } = unit;

    if (!this.inventory.has(bloodType)) {
      this.inventory.set(bloodType, []);
    }

    this.inventory.get(bloodType).push({
      unitId,
      donationDate: new Date(donationDate),
      expirationDate: new Date(expirationDate),
      status: 'available',
      tested: false
    });

    this.sortByExpiration(bloodType);
    this.checkExpirationAlert(unit);
  }

  getAvailableUnits(bloodType, quantity) {
    const units = this.inventory.get(bloodType) || [];
    const available = units.filter(unit =>
      unit.status === 'available' &&
      unit.tested === true &&
      unit.expirationDate > new Date()
    );

    return available.slice(0, quantity);
  }

  sortByExpiration(bloodType) {
    // FIFO: Use oldest blood first
    const units = this.inventory.get(bloodType);
    units.sort((a, b) => a.expirationDate - b.expirationDate);
  }

  checkExpirationAlert(unit) {
    const daysUntilExpiration = Math.ceil(
      (unit.expirationDate - new Date()) / (1000 * 60 * 60 * 24)
    );

    if (daysUntilExpiration <= 7) {
      this.expirationAlerts.push({
        unitId: unit.unitId,
        bloodType: unit.bloodType,
        daysRemaining: daysUntilExpiration,
        priority: daysUntilExpiration <= 3 ? 'urgent' : 'warning'
      });
    }
  }

  getInventoryReport() {
    const report = {};

    for (const [bloodType, units] of this.inventory.entries()) {
      const available = units.filter(u => u.status === 'available' && u.tested).length;
      const expiringSoon = units.filter(u => {
        const days = (u.expirationDate - new Date()) / (1000 * 60 * 60 * 24);
        return days <= 7 && days > 0;
      }).length;

      report[bloodType] = {
        total: units.length,
        available,
        expiringSoon,
        lastDonation: units[units.length - 1]?.donationDate
      };
    }

    return report;
  }
}
```

### 2. Donor Management
```javascript
class DonorManager {
  constructor(database) {
    this.db = database;
    this.eligibilityCriteria = {
      minAge: 18,
      maxAge: 65,
      minWeight: 50, // kg
      minDaysBetweenDonations: 56 // 8 weeks
    };
  }

  async registerDonor(donorData) {
    const { name, age, weight, bloodType, medicalHistory, contactInfo } = donorData;

    // Validate eligibility
    const eligibility = this.checkEligibility({
      age,
      weight,
      medicalHistory,
      lastDonation: null
    });

    if (!eligibility.eligible) {
      return {
        success: false,
        reason: eligibility.reason
      };
    }

    // Create donor record
    const donor = {
      id: this.generateDonorId(),
      name,
      age,
      weight,
      bloodType,
      contactInfo,
      medicalHistory,
      registrationDate: new Date(),
      donationHistory: [],
      status: 'active'
    };

    await this.db.donors.insert(donor);

    return {
      success: true,
      donorId: donor.id
    };
  }

  checkEligibility(donorInfo) {
    // Age check
    if (donorInfo.age < this.eligibilityCriteria.minAge ||
        donorInfo.age > this.eligibilityCriteria.maxAge) {
      return {
        eligible: false,
        reason: `Age must be between ${this.eligibilityCriteria.minAge} and ${this.eligibilityCriteria.maxAge}`
      };
    }

    // Weight check
    if (donorInfo.weight < this.eligibilityCriteria.minWeight) {
      return {
        eligible: false,
        reason: `Minimum weight requirement is ${this.eligibilityCriteria.minWeight}kg`
      };
    }

    // Previous donation check
    if (donorInfo.lastDonation) {
      const daysSinceLastDonation = Math.ceil(
        (new Date() - new Date(donorInfo.lastDonation)) / (1000 * 60 * 60 * 24)
      );

      if (daysSinceLastDonation < this.eligibilityCriteria.minDaysBetweenDonations) {
        return {
          eligible: false,
          reason: `Must wait ${this.eligibilityCriteria.minDaysBetweenDonations - daysSinceLastDonation} more days`,
          nextEligibleDate: new Date(
            new Date(donorInfo.lastDonation).getTime() +
            this.eligibilityCriteria.minDaysBetweenDonations * 24 * 60 * 60 * 1000
          )
        };
      }
    }

    // Medical history check
    const disqualifyingConditions = [
      'HIV', 'Hepatitis B', 'Hepatitis C', 'Malaria (recent)',
      'Recent tattoo (< 6 months)', 'Recent surgery (< 6 months)'
    ];

    const hasDisqualifyingCondition = disqualifyingConditions.some(
      condition => donorInfo.medicalHistory?.includes(condition)
    );

    if (hasDisqualifyingCondition) {
      return {
        eligible: false,
        reason: 'Medical history contains disqualifying conditions'
      };
    }

    return { eligible: true };
  }

  async scheduleDonation(donorId, appointmentDate) {
    const donor = await this.db.donors.findById(donorId);

    // Check eligibility with last donation
    const eligibility = this.checkEligibility({
      age: donor.age,
      weight: donor.weight,
      medicalHistory: donor.medicalHistory,
      lastDonation: donor.donationHistory[donor.donationHistory.length - 1]?.date
    });

    if (!eligibility.eligible) {
      return {
        success: false,
        reason: eligibility.reason,
        nextEligibleDate: eligibility.nextEligibleDate
      };
    }

    // Create appointment
    const appointment = {
      donorId,
      date: appointmentDate,
      status: 'scheduled',
      createdAt: new Date()
    };

    await this.db.appointments.insert(appointment);

    // Send confirmation
    await this.sendAppointmentConfirmation(donor, appointment);

    return {
      success: true,
      appointmentId: appointment.id
    };
  }
}
```

```javascript
class DistributionManager {
  constructor(inventory, database) {
    this.inventory = inventory;
    this.db = database;
  }

  async createBloodRequest(requestData) {
    const { hospitalId, bloodType, quantity, urgency, patientInfo } = requestData;

    // Check availability
    const availableUnits = this.inventory.getAvailableUnits(bloodType, quantity);

    if (availableUnits.length < quantity) {
      return {
        success: false,
        reason: 'Insufficient blood units available',
        available: availableUnits.length,
        requested: quantity
      };
    }

    // Create request
    const request = {
      id: this.generateRequestId(),
      hospitalId,
      bloodType,
      quantity,
      urgency,
      patientInfo,
      status: 'pending',
      createdAt: new Date(),
      allocatedUnits: availableUnits.map(u => u.unitId)
    };

    await this.db.requests.insert(request);

    // Reserve units
    availableUnits.forEach(unit => {
      unit.status = 'reserved';
      unit.reservedFor = request.id;
    });

    return {
      success: true,
      requestId: request.id,
      estimatedDelivery: this.calculateDeliveryTime(urgency)
    };
  }

  async fulfillRequest(requestId) {
    const request = await this.db.requests.findById(requestId);

    if (!request || request.status !== 'pending') {
      return { success: false, reason: 'Invalid request' };
    }

    // Mark units as distributed
    request.allocatedUnits.forEach(unitId => {
      const unit = this.findUnitById(unitId);
      unit.status = 'distributed';
      unit.distributedTo = request.hospitalId;
      unit.distributionDate = new Date();
    });

    // Update request status
    request.status = 'fulfilled';
    request.fulfilledAt = new Date();
    await this.db.requests.update(requestId, request);

    // Generate distribution report
    await this.generateDistributionReport(request);

    return {
      success: true,
      distributionId: this.generateDistributionId()
    };
  }

  calculateDeliveryTime(urgency) {
    const baseTime = new Date();

    switch(urgency) {
      case 'critical':
        baseTime.setMinutes(baseTime.getMinutes() + 30);
        break;
      case 'urgent':
        baseTime.setHours(baseTime.getHours() + 2);
        break;
      case 'routine':
        baseTime.setHours(baseTime.getHours() + 24);
        break;
    }

    return baseTime;
  }
}
```

### 4. Blood Testing & Quality Control
```javascript
class TestingModule {
  constructor(database) {
    this.db = database;
    this.requiredTests = [
      'Blood Type Verification',
      'HIV',
      'Hepatitis B',
      'Hepatitis C',
      'Syphilis',
      'Malaria'
    ];
  }

  async performTests(unitId) {
    const testResults = {
      unitId,
      testDate: new Date(),
      results: {},
      overallStatus: 'pending'
    };

    // Simulate testing process
    for (const test of this.requiredTests) {
      testResults.results[test] = {
        result: 'negative', // In real system, this would come from lab
        testedBy: 'Lab Technician',
        timestamp: new Date()
      };
    }

    // Determine overall status
    const allNegative = Object.values(testResults.results)
      .every(result => result.result === 'negative');

    testResults.overallStatus = allNegative ? 'safe' : 'rejected';

    // Update unit status
    await this.db.bloodUnits.update(unitId, {
      tested: true,
      testResults,
      status: allNegative ? 'available' : 'rejected'
    });

    return testResults;
  }
}
```
